63 if (beaconIntentions == null) { return; }
64
65#if UNITY_EDITOR
66 if (updateOrder == 0)
67 {
68 Debug.LogWarning("updateOrder is neither set to Before nor After on " + this + " : the intentions will never be updated.");
69 }
70#endif
71
72 if ((updateOrder & UpdateOrder.Before) != 0)
73 beaconIntentions.Tick();
74
75 float3 position = transform.position, goal = beaconIntentions.goalLocation;
76 float3 heading = normalize(goal - position);
77 float sqDistToGoal = distancesq(position, goal);
78
79 float alteredSpeed = speed;
80
81 //Check if we should sample dropoff pattern based on current distance to goal.
82 if (dropOffDistance > 0f && dropOffPattern != null)
83 {
84 float sqDropOff = lengthsq(dropOffDistance);
85 if (sqDistToGoal <= sqDropOff)
86 {
87 alteredSpeed = speed * dropOffPattern.Evaluate(1f - sqDistToGoal / sqDropOff);
88 }
89 }
90
91 float3 velocity = heading * alteredSpeed;
92 IntentionState iState = beaconIntentions.currentState;
93
94 if (beaconProcessors != null)
95 {
96 BeaconProcessor processor;
97 for (int i = 0, count = beaconProcessors.Length; i < count; i++)
98 {
99 processor = beaconProcessors[i];
100 if (processor == null
101 || !processor.enabled
102 || (processor.allowedStates & iState) == 0)
103 {
104 continue;
105 }
106
107 processor.Apply(
108 ref position,
109 ref velocity,
110 ref heading,
111 sqDistToGoal);
112 }
113 }
114
115 transform.position = position;
116
117 if (lookAt != LookAt.Nothing)
118 {
119 float3 dir = float3(0f);
120
121 if (lookAt == LookAt.Goal)
122 dir = normalize(goal - position);
123 else if (lookAt == LookAt.Direction)
124 dir = heading;
125 else if (lookAt == LookAt.CustomTarget && lookAtTarget != null)
126 dir = normalize((float3)lookAtTarget.position - position);
127
128 if (plane == AxisPair.XY)
129 transform.up = float3(dir.x, dir.y, 0f);
130 else
131 transform.forward = dir;
132 }
133
134 if ((updateOrder & UpdateOrder.After) != 0)
135 beaconIntentions.Tick();
136
137 }
